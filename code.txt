#include <Keypad.h>
#include <ESP32Servo.h>

#define SERVO_PIN 14 // connect servo signal pin
#define buzz 4   // connect buzzer positive pin 
Servo myServo;

const byte ROWS = 4;
const byte COLS = 4;

char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

byte rowPins[ROWS] = {18, 19, 22, 23}; // connect rows pins
byte colPins[COLS] = {2, 15, 12, 13};  // connect column pins

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

String password = "1234";   // Set your password
String inputPassword = "";

void setup() {
  Serial.begin(115200);
  
  myServo.attach(SERVO_PIN);
  myServo.write(0);  // Locked position
  
  Serial.println("Smart Door Lock System Ready");
}

void loop() {
  char key = keypad.getKey();
  
  if (key) {
    Serial.print("Key Pressed: ");
    Serial.println(key);

    if (key == '#') {   // Enter key
      if (inputPassword == password) {
        Serial.println("Access Granted");
        unlockDoor();
      } else {
        
        Serial.println("Access Denied");
      }
      inputPassword = "";
    }
    else if (key == '*') {   // Clear key
      inputPassword = "";
      Serial.println("Password Cleared");
    }
    else {
      inputPassword += key;
    }
  }
}

void unlockDoor() {
  myServo.write(160);   // Unlock
  delay(5000);         // Door open for 5 seconds
  myServo.write(0);    // Lock again
}
